123bool Graph::bfs(int v, int dest) {
124
125 for(int i = 1; i <= n; i++) {
126 nodes[i].visited = false;
127 nodes[i].pred = -1;
128 }
129
130 queue<int> q;
131
132 q.push(v);
133 nodes[v].visited = true;
134
135 while (!q.empty()) {
136 int tmp = q.front(); q.pop();
137
138 for (const auto& e: nodes[tmp].adj) {
139 int w = e.dest;
140 if (!nodes[w].visited) {
141 q.push(w);
142 nodes[w].visited = true;
143 nodes[w].pred = tmp;
144
145 if (tmp == dest) return true;
146 }
147 }
148 }
149 return false;

201void Graph::dijkstra(int s, int b) {
202 MinHeap<int,double> q(n, -1);
203
204 for (int v = 1; v <= n; v++) {
205 nodes[v].dist = INFINITE;
206 q.insert(v, INFINITE);
207 nodes[v].visited = false;
208 nodes[v].pred = -1;
209 }
210
211 nodes[s].dist = 0;
212 q.decreaseKey(s, 0);
213 nodes[s].pred = s;
214
215 while (q.getSize() > 0) {
216 int u = q.removeMin();
217 nodes[u].visited = true;
218 if (u == b) return;
219
220 for (const auto& elem : nodes[u].adj) {
221
222 int e = elem.dest;
223 double w = elem.weight;
224
225 if (!nodes[e].visited && (nodes[u].dist + w < nodes[e].dist)) {
226 nodes[e].dist = nodes[u].dist + w;
227 q.decreaseKey(e, nodes[e].dist);
228 nodes[e].pred = u;
229 }
230 }
231 }
