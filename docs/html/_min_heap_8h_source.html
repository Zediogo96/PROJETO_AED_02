111template <class K, class V>
112void MinHeap<K,V>::upHeap(int i) {
113 while (i>1 && a[i].value < a[PARENT(i)].value) { // while pos smaller than parent, keep swapping to upper position
114 swap(i, PARENT(i));
115 i = PARENT(i);
116 }
117}

119template <class K, class V>
120void MinHeap<K,V>::downHeap(int i) {
121 while (LEFT(i) <= size) { // while within heap limits
122 int j = LEFT(i);
123 if (RIGHT(i)<=size && a[RIGHT(i)].value < a[j].value) j = RIGHT(i); // choose smaller child
124 if (a[i].value < a[j].value) break; // node already smaller than children, stop
125 swap(i, j); // otherwise, swap with smaller child
126 i = j;
127 }
128}

131void MinHeap<K,V>::swap(int i1, int i2) {
132 Node tmp = a[i1]; a[i1] = a[i2]; a[i2] = tmp;
133 pos[a[i1].key] = i1;
134 pos[a[i2].key] = i2;

168void MinHeap<K,V>::decreaseKey(const K& key, const V& value) {
169 if (!hasKey(key)) return; // key does not exist, do nothing
170 int i = pos[key];

172 a[i].value = value;

177template <class K, class V>
178K MinHeap<K,V>::removeMin() {
179 if (size == 0) return KEY_NOT_FOUND;
180 K min = a[1].key;
181 pos.erase(min);
182 a[1] = a[size--];
183 downHeap(1);
